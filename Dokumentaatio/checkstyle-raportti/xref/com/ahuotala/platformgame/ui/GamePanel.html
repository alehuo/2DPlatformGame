18  package com.ahuotala.platformgame.ui;
19  
20  import com.ahuotala.platformgame.Game;
21  import com.ahuotala.platformgame.entity.Player;
22  import com.ahuotala.platformgame.level.GameLevel;
23  import java.awt.Color;
24  import java.awt.Font;
25  import java.awt.Graphics;
26  import javax.swing.JPanel;
27  
28  /**
29   * Pelipaneeli piirtää tason, entiteetit sekä pelaajan.
30   *
31   * @author ahuotala
32   */
33  public class GamePanel extends JPanel {
34  
35      //SerialVersionUID
36      private static final long serialVersionUID = 1L;
37  
38      //Taso
39      private GameLevel level;
40  
41      /**
42       * Pelipaneelin konstruktori.
43       */
44      public GamePanel() {
45          super();
46      }
47  
48      /**
49       * Aseta pelitaso pelipaneeliin, jotta sen entiteetit voidaan piirtää
50       * näytölle (ja päivittää niitä).
51       *
52       * @param level pelitaso
53       */
54      public void setLevel(GameLevel level) {
55          this.level = level;
56      }
57  
58      /**
59       * Tätä käytetään piirtämiseen.
60       *
61       * @param g
62       */
63      @Override
64      public void paintComponent(Graphics g) {
65  
66          super.paintComponent(g);
67  
68          //Tausta
69          Color c = new Color(93, 148, 251);
70          g.setColor(c);
71          g.fill3DRect(0, 0, Game.WINDOWWIDTH, Game.WINDOWHEIGHT, false);
72  
73          //Taso
74          if (level != null) {
75              level.getTiles().forEach((tile) -> {
76                  tile.render(g);
77              });
78              level.getEntities().forEach((entity) -> {
79                  entity.render(g);
80              });
81          }
82  
83          Player player = level.getPlayer();
84  
85          //Pelaaja
86          if (player != null) {
87              player.render(g);
88              g.setColor(Color.WHITE);
89  
90              //Ohjeet
91              g.drawString("[A] Vasen  [D] Oikea  [Space]  Hyppää", 4, 20);
92  
93              //Debuggausta varten x:t ja y:t jne..
94              g.drawString("playerX: " + (player.getX() - Game.STARTINGOFFSET + Player.offsetX), 4, 35);
95              g.drawString("playerY: " + player.getY(), 4, 50);
96              g.drawString("isFalling: " + player.isFalling(), 4, 65);
97              g.drawString("walkingDirection: " + player.getWd(), 4, 80);
98              g.drawString("playerBoundsX: " + (player.getBounds().getX()), 4, 95);
99              g.drawString("playerBoundsY: " + (player.getBounds().getY()), 4, 110);
100             //Pisteytys
101             Font f = g.getFont();
102             Font newF = f.deriveFont(f.getSize() * 3F);
103             g.setFont(newF);
104             g.drawString("Score: " + level.getScore().getValue(), Game.WINDOWWIDTH / 2, 32);
105             g.drawString("Time: " + level.getScore().getCurrentTime() / 1000 + " s", Game.WINDOWWIDTH / 2, 70);
106         }
107 
108         //Todo..
109     }
110 
111     /**
112      * Antaa päivityskutsun tasolle.
113      */
114     public void tick() {
115         level.tick();
116     }
117 }
